function register(event) {

const div = event.currentTarget.parentNode;

const name = div.children.name.value;

const password = div.children.password.value;

const passwordConfirm = div.children.passwordConfirm.value;


if (password === passwordConfirm) {

alert(`successfully registered!\n\nWelcome: ${name}`);

} else {

alert(`Invalid Passwords.\n\nTry Again!`);

}

}


const btnRegister = document.getElementById("registerBtn");

function removeEvent() {

Usar o metodo preventDefault() para conseguirmos ver os nossos items. Sem isto a página ia atualizar por causa do comportamento default do form e não conseguiamos ver os items

Criamos uma variavel que faz ligação com o valor do nosso input name, fizemos ligação com o querySelector, referindo o input com [name="x"]

Criamos uma variavel que faz ligação com o valor da nossa textarea observations, fizemos ligação com o querySelector, referindo a textarea[name="b"]

Criamos uma variavel com uma string vazia

Usamos um querySelectorAll para selecionar o valor de todos os inputs que levarem CHECK (.checked), referindo os inputs com [name="c"], usamos o metodo forEach()para termos acesso a esses items, usando uma função anonima dentro do .forEach com parametro: .forEach(function(item){)

Agora usamos a variavel que criamos em cima "salad" e usámos essa variavel para aceder a todos os valores desses items: salad += `${item.value}

const form = document.getElementById("orderForm");

form.addEventListener("submit", function (ev) {

ev.preventDefault();

const name = document.querySelector("input[name='name']").value;

const address = document.querySelector("input[name='address']").value;

const breadType = document.querySelector("select[name='breadType']").value;

const main = document.querySelector("input[name='main']").value;

const observations = document.querySelector("textarea[name='observations']").value;

let salad = "";

document.querySelectorAll("input[name='salad']:checked").forEach(function (item) {

salad += `- ${item.value} \n`;

});

console.log({name,address,breadType,main,salad,observations,});

alert(`Pedido realizado!\n\nNome do cliente: ${name}\nMorada de entrega: ${address}\nTipo de pão: ${breadType}\nRecheio: ${main}\nSalada: ${salad}\nObservações: ${observations}`);

});


Dica: Usando o metodo "preventDefault" -> Conseguimos prevenir o comportamento default do nosso evento. Neste caso no formulário. Não deixa executar a action default dele.

Com isso, Conseguimos mostrar os valores dos nossos items. Sem isso, o browser atualiza muito rapido a pagina e não conseguimos ver nada

Funciona assim: parametro.preventDefault()

No nosso caso: event.preventDefault()

Recursos de Armazenamento do Navegador


Session Storage:

Forma simples de armazenar dados no navegador durante a navegação

Guarda as informações mesmo depois de mudarmos de página num site enquanto a janela estiver aberta

Guarda os dados daquele site, enquanto a janela está aberta, ao fechahr o sessionStorage é limpo



Local Storage:

Dados armazenados no navegador ficam guardados mesmo depois de fecharmos a janela do navegador

Existem "para sempre" até serem removidos, não possuem data de expiração/p>

Cookies:

Dados armazenados em pequenos arquivos de texto no computador do cliente

Podem possuir uma data de expiração

A forma mais tradicional de permitir "lembrar" informações mesmo depois de fechar o navegador

São enviados nas requisições e podem existir no servidor, sendo devolvidos quando uma pagina html for solicitada
